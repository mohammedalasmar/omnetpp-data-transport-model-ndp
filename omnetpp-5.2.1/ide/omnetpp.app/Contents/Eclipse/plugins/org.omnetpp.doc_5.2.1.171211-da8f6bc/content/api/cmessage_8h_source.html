670 template<typename T>
671 inline int cMessage::findTag()
672 {
673  if (!tags)
674  return -1;
675  int n = tags->size();
676  for (int i = 0; i < n; i++) {
677  cObject *tag = (*tags)[i];
678  if (typeid(T) == typeid(*tag)) //TODO this could be regular (non-template) method and go out-of-line if it took typeid(T) as parameter
679  return i;
680  }
681  return -1;
682 }
