265 inline long double_to_long(double d)
266 {
267  // gcc feature: if double d=0xc0000000, (long)d yields 0x80000000 !
268  // This only happens with long: unsigned long is OK.
269  // This causes trouble if we in fact want to cast this long to unsigned long, see NED_expr_2.test.
270  // Workaround follows. Note: even the ul variable is needed: when inlining it, gcc will do the wrong cast!
271  long l = (long)d;
272  unsigned long ul = (unsigned long)d;
273  return d<0 ? l : ul;
274 }
